library(tidyverse)
library(tidymodels)
library(cfbfastR)
library(janitor)
library(bonsai)
library(gt)

set.seed(1001)


wrdraftedstats <- read_csv("https://mattwaite.github.io/sportsdatafiles/wrdraftedlaststats20132022.csv")

ratings <- read_csv("cfbratings.csv") %>% mutate(draft_year = year+1)

withrating <- wrdraftedstats %>% inner_join(ratings, by=c("college_team"="team", "college_conference"="conference", "year"="draft_year"))

wrselected <- withrating %>%
  select(
    name,
    year,
    college_team,
    nfl_team,
    overall,
    receiving_rec,
    receiving_ypr,
    receiving_td,
    receiving_long,
    offense_ranking,
    PPR
  ) %>% na.omit

player_split <- initial_split(wrselected, prop = .8)
player_train <- training(player_split)

player_test <- testing(player_split)

player_recipe <-
  recipe(PPR ~ ., data = player_train) %>%
  update_role(name, year, college_team, nfl_team, new_role = "ID")

rf_mod <-
  rand_forest() %>%
  set_engine("ranger") %>%
  set_mode("regression")

rf_workflow <-
  workflow() %>%
  add_model(rf_mod) %>%
  add_recipe(player_recipe)

rf_fit <-
  rf_workflow %>%
  fit(data = player_train)

rfpredict <-
  rf_fit %>%
  predict(new_data = player_train) %>%
  bind_cols(player_train)



rftestpredict <-
  rf_fit %>%
  predict(new_data = player_test) %>%
  bind_cols(player_test)




currentwrstats <- read_csv("https://mattwaite.github.io/sportsdatafiles/currentwrstats.csv")

currentdraft <- cfbd_draft_picks(2023) %>% filter(position == "Wide Receiver")

stats22 <- read_csv("stats22.csv")


currentwithstats <- currentdraft %>% inner_join(stats22, by=c("name" = "player"))

currentwrdraftclass <- currentwithstats %>% inner_join(ratings, by=c("college_team"="team", "year.x"="draft_year"))

currentwrselected <- currentwrdraftclass %>%
  select(
    name,
    year,
    college_team,
    nfl_team,
    overall,
    receiving_rec,
    receiving_ypr,
    receiving_td,
    receiving_long,
    offense_ranking,
  ) %>% na.omit()

currentrfpredict <- 
  rf_fit %>% 
  predict(new_data = currentwrselected) %>%
  bind_cols(currentwrselected)

currentrfpredict %>% 
  top_n(5, wt=.pred) %>%
  select(name, nfl_team, overall, .pred) %>%
  arrange(desc(.pred)) %>%
  gt() %>%
  cols_label(
    name = "Name",
    nfl_team = "NFL Team",
    overall = "Draft Pick",
    .pred = "Predicted Fantasy Points"
  ) %>%
  tab_header(
    title = "Which Rookie WRs will score the most Fantasy Points in 2023?",
    subtitle = "The earliest drafted WRs will not necessarily have the best seasons."
  ) %>%  
  tab_source_note(
    source_note = md("**By:** Jack Yelden")
  ) %>% 
  tab_style(
    style = cell_text(color = "black", weight = "bold", align = "left"),
    locations = cells_title("title")
  ) %>% 
  tab_style(
    style = cell_text(color = "black", align = "left"),
    locations = cells_title("subtitle")
  ) %>%

     locations = cells_column_labels(columns = everything()),
